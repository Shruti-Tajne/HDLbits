count4 the_counter (.clk(clk), .enable(c_enable), .load(c_load), .d(c_d) , .Q(Q));
    
    assign c_enable = enable;
    
    always@(*) begin
        if (reset) begin
            c_load = 1;
            c_d = 1;
        end
        else begin
            c_load = 0;
            c_d = 0;
            if (enable && Q==4'd12) begin
            	c_load = 1;
                c_d = 1;
            end
        end
    end
